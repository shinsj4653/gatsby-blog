보간법

let myName = 'HEROPY!'
let hello = `Hello ${myName}`

console.log(hello) //Hello HEROPY!
  • ${ } 중괄호 안에 변수를 넣어 문자열로 변환시키는 역할
  • 보간법 사용하려면 문자열을 Backtick( ` ) 기호안에 작성해야함

함수

  • 기명(이름이 있는) 함수
  • 함수 선언!
function hello() {
  console.log('Hello~')
}
  • 익명(이름이 없는) 함수
  • 함수 표현!
const world = function () {
  console.log('World~')
}
  • 함수 호출
hello() // 기명함수 호출: Hello~
world() // 익명함수 호출: World~
